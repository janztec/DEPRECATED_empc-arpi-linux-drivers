# I2C
dtparam=i2c_arm=on

# SPI
dtparam=spi=on

# SERIAL CONSOLE
enable_uart=1

# CAN
dtoverlay=mcp2515-can0,oscillator=16000000,interrupt=25

# SERIAL PORT RS232/RS485
# if serial tranceiver mode jumper J301 is:
#   SET (RS232)    : use overlay: sc16is7xx-ttysc0-rs232
#   NOT SET (RS485): use overlay: sc16is7xx-ttysc0-rs485
dtoverlay=sc16is7xx-ttysc0-rs232

# enable more than 600mA for USB ports
max_usb_current=1

[pi3]
# forces fixed 600MHz CPU frequency (see manual)
arm_freq=600

# green LED (gpio5) as SD card activity
dtoverlay=pi3-act-led,gpio=5,activelow=off

# disables Bluetooth and restores UART0/ttyAMA0
dtoverlay=pi3-miniuart-bt

[pi2]
# green LED (gpio5) as SD card activity
dtparam=act_led_gpio=5

# selects the bcm2835-sdhost SD/MMC driver
dtoverlay=sdhost

[all]
